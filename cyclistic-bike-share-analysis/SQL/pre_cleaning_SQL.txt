-- Step 1: Inspect Dataset Structure
-- Purpose: Verify column data types and nullability

-- I began by inspecting the schema of the consolidated dataset to ensure that all columns were stored in appropriate formats and to identify if any fields might require type conversion or special handling.

SELECT 
  column_name,
  data_type,
  is_nullable
FROM `driven-realm-460621-r2.cyclistic_project.INFORMATION_SCHEMA.COLUMNS`
WHERE table_name = 'full_year_tripdata'

Result: No signs of inconcistencies

-- -----------------------------------------------------

-- Step 2: Validate Rideable Types
-- Purpose: Confirm that rideable_type values align with expected categories

-- I explored the distinct ride types to ensure consistency with Cyclistic’s offerings and to verify that no unexpected categories were present.

SELECT DISTINCT rideable_type
FROM `driven-realm-460621-r2.cyclistic_project.full_year_tripdata`

-- Result: Only the expected categories were present: classic_bike, electric_bike, electric_scooter


-- -----------------------------------------------------

-- Step 3: Validate User Segmentation
-- Purpose: Confirm member vs casual classification

-- I checked the distinct values in the member_casual column to ensure proper segmentation between casual riders and annual members.

SELECT DISTINCT member_casual
FROM `driven-realm-460621-r2.cyclistic_project.full_year_tripdata`

-- Result: Two categories identified: 'casual' and 'member'


-- -----------------------------------------------------

-- Step 4: Validate Ride ID Consistency
-- Purpose: Ensure ride_id format is consistent

-- I examined the length distribution of ride_id values. 
-- Consistent length indicates standardised formatting.

SELECT 
  LENGTH(ride_id) AS id_length, 
  COUNT(*) AS count
FROM `driven-realm-460621-r2.cyclistic_project.full_year_tripdata`
GROUP BY id_length

-- Result: All ride_id values are 16 characters long


-- -----------------------------------------------------

-- Step 5: Check for Duplicate Ride IDs
-- Purpose: Ensure each trip record is unique since duplicate IDs could lead to incorrect aggregations and a skewed analysis.

SELECT 
  ride_id, 
  COUNT(*) AS count
FROM `driven-realm-460621-r2.cyclistic_project.full_year_tripdata`
GROUP BY ride_id
HAVING COUNT(*) > 1
ORDER BY count DESC

-- Result: No duplicate ride_id values found


-- -----------------------------------------------------

-- Step 6: Check for Missing Values in the dataset
-- Purpose: Identify null values across key fields

-- This step helped assess data completeness and identify rows that may require cleaning or exclusion.

SELECT
  COUNTIF(ride_id IS NULL) AS missing_ride_id,
  COUNTIF(rideable_type IS NULL) AS missing_rideable_type,
  COUNTIF(started_at IS NULL) AS missing_started_at,
  COUNTIF(ended_at IS NULL) AS missing_ended_at,
  COUNTIF(start_station_name IS NULL) AS missing_start_station_name,
  COUNTIF(start_station_id IS NULL) AS missing_start_station_id,
  COUNTIF(end_station_name IS NULL) AS missing_end_station_name,
  COUNTIF(end_station_id IS NULL) AS missing_end_station_id,
  COUNTIF(start_lat IS NULL) AS missing_start_lat,
  COUNTIF(start_lng IS NULL) AS missing_start_lng,
  COUNTIF(end_lat IS NULL) AS missing_end_lat,
  COUNTIF(end_lng IS NULL) AS missing_end_lng,
  COUNTIF(member_casual IS NULL) AS missing_member_casual
FROM `driven-realm-460621-r2.cyclistic_project.full_year_tripdata`

-- Result: Missing values were identified primarily in:
-- start_station_name, end_station_name, station IDs, and coordinate fields


-- -----------------------------------------------------

-- Step 7: Validate Geographic Coordinates
-- Purpose: Ensure latitude and longitude values are within valid ranges

SELECT *
FROM `driven-realm-460621-r2.cyclistic_project.full_year_tripdata`
WHERE
  start_lat NOT BETWEEN -90 AND 90 OR
  end_lat NOT BETWEEN -90 AND 90 OR
  start_lng NOT BETWEEN -180 AND 180 OR
  end_lng NOT BETWEEN -180 AND 180

-- Result: No invalid geographic coordinates detected


-- -----------------------------------------------------

-- Step 8: Identify Invalid Time Sequences
-- Purpose: Detect trips where end time precedes start time

-- These records indicate data inconsistencies and must be removed.

SELECT *
FROM `driven-realm-460621-r2.cyclistic_project.full_year_tripdata`
WHERE ended_at < started_at

-- Result: Invalid records identified and flagged for removal


-- -----------------------------------------------------

-- Step 9: Identify Duration Outliers
-- Purpose: Detect unrealistic trip durations

-- Trips shorter than 1 minute or longer than 24 hours were considered as anomalies.

SELECT 
  ride_id,
  member_casual,
  started_at,
  ended_at,
  TIMESTAMP_DIFF(ended_at, started_at, MINUTE) AS duration_min
FROM `driven-realm-460621-r2.cyclistic_project.full_year_tripdata`
WHERE 
  TIMESTAMP_DIFF(ended_at, started_at, MINUTE) < 1
  OR 
  TIMESTAMP_DIFF(ended_at, started_at, MINUTE) > 1440
ORDER BY duration_min DESC

-- Finding: 131,461 trips fell outside the valid duration range 
-- and were excluded during the data cleaning process
